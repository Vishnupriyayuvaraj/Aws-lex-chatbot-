# Route intent to the correct handler
def lambda_handler(event, context):
    intent_name = event['sessionState']['intent']['name']

    HANDLERS = {
        'WelcomeIntent':  welcome,
        'CheckBalance':   check_balance,
        'TransferFunds':  transfer_funds,
        'FallbackIntent': fallback,
    }
    return HANDLERS[intent_name](event)

# Return a mock balance after slot fulfillment
def check_balance(event):
    account_type = get_slot(event, 'accountType')
    balance      = get_mock_balance(account_type)

    return close(event, {}, 'Fulfilled', {
        'contentType': 'PlainText',
        'content': f"Your {account_type} balance is ${balance:,.2f}."
    })
